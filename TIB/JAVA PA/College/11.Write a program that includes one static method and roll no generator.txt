11.Write a program that includes atleast one static method and atleast one basic method.
Display the college name, student name and student roll. The static method will accept the
college name and the other method will accept the Name of the student. The roll will not be
user defined, it will be auto generated by incrementing the roll number everytime a new
student object is created. 
----------------------------------------------------------------
import java.util.Scanner;

class Student {
    private static int nextRollNumber = 1001; // Starting roll number
    private String collegeName;
    private String studentName;
    private int rollNumber;

    
    static void setCollegeName(String college) {
        System.out.println("Setting College Name: " + college);
    }

   
    void setStudentDetails(String name) {
        this.studentName = name;
        this.rollNumber = nextRollNumber++;
    }

   
    void displayStudentInfo() {
        System.out.println("College Name: " + collegeName);
        System.out.println("Student Name: " + studentName);
        System.out.println("Roll Number: " + rollNumber);
    }
}

public class CollegeStudentProgram {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

       
        Student.setCollegeName("Example College");

       
        Student student1 = new Student();

       
        System.out.print("Enter the student name: ");
        String studentName = scanner.nextLine();
        student1.setStudentDetails(studentName);

        
        System.out.println("\nStudent Information:");
        student1.displayStudentInfo();

        
        Student student2 = new Student();

        
        System.out.print("\nEnter the name of another student: ");
        String studentName2 = scanner.nextLine();
        student2.setStudentDetails(studentName2);

        
        System.out.println("\nStudent Information (Second Student):");
        student2.displayStudentInfo();

        scanner.close();
    }
}
