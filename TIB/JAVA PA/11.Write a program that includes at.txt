11.Write a program that includes atleast one static method and atleast one basic method.
Display the college name, student name and student roll. The static method will accept the
college name and the other method will accept the Name of the student. The roll will not be
user defined, it will be auto generated by incrementing the roll number everytime a new
student object is created.

class Student {
    static String collegeName;

    String studentName;
    int roll;

    static void setCollegeName(String name) {
        collegeName = name;
    }

    void setStudentName(String name) {
        studentName = name;
    }

    void displayDetails() {
        roll++;
        System.out.println("College Name: " + collegeName);
        System.out.println("Student Name: " + studentName);
        System.out.println("Student Roll: " + roll);
    }

    public static void main(String[] args) {
        setCollegeName("ABC College");
        
        Student student1 = new Student();
        student1.setStudentName("John");
        student1.displayDetails();

        Student student2 = new Student();
        student2.setStudentName("Jane");
        student2.displayDetails();
    }
}
